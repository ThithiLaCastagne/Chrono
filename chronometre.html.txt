<script>
/* =========================================
   CHRONOMÈTRE CLASSIQUE
   ========================================= */
var running = false, startTime = 0, elapsed = 0, rafId = null, lapStart = 0, laps = [];

function pad(n, d) { return String(Math.floor(n)).padStart(d || 2, '0'); }

function fmtMs(ms) {
  var s  = Math.floor(ms / 1000);
  var h  = Math.floor(s / 3600);
  var m  = Math.floor((s % 3600) / 60);
  var sc = s % 60;
  var t  = Math.floor((ms % 1000) / 100);
  return pad(h) + ':' + pad(m) + ':' + pad(sc) + '.' + t;
}

function mainAction() {
  if (!running) {
    running   = true;
    startTime = Date.now() - elapsed;
    lapStart  = startTime;
    document.getElementById('btn-main').textContent = 'Arrêter';
    document.getElementById('btn-main').className   = 'btn-ios btn-red';
    document.getElementById('btn-sec').textContent  = 'Tour';
    tickChrono();
  } else {
    running = false;
    cancelAnimationFrame(rafId);
    elapsed = Date.now() - startTime;
    document.getElementById('btn-main').textContent = 'Démarrer';
    document.getElementById('btn-main').className   = 'btn-ios btn-green';
    document.getElementById('btn-sec').textContent  = 'Réinitialiser';
  }
}

function secAction() {
  if (running) {
    var now = Date.now();
    laps.unshift({ n: laps.length + 1, t: fmtMs(now - lapStart) });
    lapStart = now;
    renderLaps();
  } else if (elapsed > 0) {
    elapsed = 0; laps = [];
    document.getElementById('time-big').textContent  = '00:00:00.0';
    document.getElementById('time-mini').textContent = '00:00:00.0';
    document.getElementById('btn-sec').textContent   = 'Tour';
    renderLaps();
  }
}

function tickChrono() {
  var ms = Date.now() - startTime;
  var s  = fmtMs(ms);
  document.getElementById('time-big').textContent  = s;
  document.getElementById('time-mini').textContent = s;
  if (running) rafId = requestAnimationFrame(tickChrono);
}

function renderLaps() {
  var el = document.getElementById('laps-area');
  if (!laps.length) { el.innerHTML = '<div class="lap-empty">Aucun tour</div>'; return; }
  el.innerHTML = laps.map(function(l) {
    return '<div class="lap-row"><span class="lap-label">Tour ' + l.n + '</span><span class="lap-time">' + l.t + '</span></div>';
  }).join('');
}

/* =========================================
   MULTI CHRONOMÈTRES
   ========================================= */
var chronos = [];
try { chronos = JSON.parse(localStorage.getItem('ck_chronos') || '[]'); } catch(e) {}

function save() { try { localStorage.setItem('ck_chronos', JSON.stringify(chronos)); } catch(e) {} }

function pad2(n) { return String(Math.floor(n)).padStart(2, '0'); }

function fmtElapsed(ms) {
  var s  = Math.floor(Math.max(0, ms) / 1000);
  var yr = Math.floor(s / 31557600);
  var mo = Math.floor((s % 31557600) / 2629800);
  var d  = Math.floor((s % 2629800)  / 86400);
  var h  = Math.floor((s % 86400)    / 3600);
  var m  = Math.floor((s % 3600)     / 60);
  var sc = s % 60;
  if (yr > 0) return yr + 'a ' + mo + 'm ' + d + 'j';
  if (mo > 0) return mo + 'm ' + d + 'j ' + pad2(h) + 'h';
  if (d  > 0) return d  + 'j ' + pad2(h) + ':' + pad2(m) + ':' + pad2(sc);
  return pad2(h) + ':' + pad2(m) + ':' + pad2(sc);
}

function fmtSince(iso) {
  var d = new Date(iso);
  return 'depuis le ' + d.toLocaleDateString('fr-FR', { day: '2-digit', month: 'long', year: 'numeric' })
    + ' à ' + d.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' });
}

function renderCards() {
  var el = document.getElementById('chrono-cards');
  if (!chronos.length) {
    el.innerHTML = '<div class="multi-empty">Aucun chronomètre.<br>Appuie sur <strong style="color:#5ac85a">+ Nouveau</strong><br>pour en ajouter un.</div>';
    return;
  }
  el.innerHTML = chronos.map(function(c, i) {
    var ms = Math.max(0, Date.now() - new Date(c.start).getTime());
    return '<div class="chrono-item">'
      + '<div class="ci-name">' + c.label + '</div>'
      + '<div class="ci-time" id="cv' + i + '">' + fmtElapsed(ms) + '</div>'
      + '<div class="ci-sub"  id="cs' + i + '">' + fmtSince(c.start) + '</div>'
      + '<button class="ci-del" onclick="delChrono(' + i + ')">Supprimer</button>'
      + '</div>';
  }).join('');
}

function tickCards() {
  chronos.forEach(function(c, i) {
    var el = document.getElementById('cv' + i);
    if (!el) return;
    el.textContent = fmtElapsed(Math.max(0, Date.now() - new Date(c.start).getTime()));
  });
}

function openAdd() {
  var f = document.getElementById('add-form');
  f.classList.add('open');
  var now = new Date();
  now.setMinutes(now.getMinutes() - now.getTimezoneOffset());
  document.getElementById('inp-dt').value = now.toISOString().slice(0, 16);
  document.getElementById('inp-name').focus();
}

function closeAdd() {
  document.getElementById('add-form').classList.remove('open');
}

function addChrono() {
  var name = document.getElementById('inp-name').value.trim();
  var dt   = document.getElementById('inp-dt').value;
  if (!name || !dt) { alert('Remplis le nom et la date !'); return; }
  chronos.push({ label: name, start: dt });
  save();
  renderCards();
  closeAdd();
  document.getElementById('inp-name').value = '';
}

function delChrono(i) {
  if (!confirm('Supprimer "' + chronos[i].label + '" ?')) return;
  chronos.splice(i, 1);
  save();
  renderCards();
}

/* =========================================
   NAVIGATION
   ========================================= */
function switchTab(tab) {
  document.querySelectorAll('.view').forEach(function(v) { v.classList.remove('active'); });
  document.querySelectorAll('.nav-item').forEach(function(b) { b.classList.remove('active'); });
  if (tab === 'chrono') {
    document.getElementById('view-chrono').classList.add('active');
    document.querySelectorAll('.nav-item')[0].classList.add('active');
  } else {
    document.getElementById('view-multi').classList.add('active');
    document.getElementById('tab-multi').classList.add('active');
    renderCards();
  }
}

/* =========================================
   HORLOGE BARRE DE STATUT
   ========================================= */
function updateClock() {
  var now = new Date();
  document.getElementById('sb-time').textContent =
    now.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' });
}

/* =========================================
   INIT
   ========================================= */
updateClock();
setInterval(updateClock, 10000);
setInterval(tickCards, 1000);
switchTab('multi');
